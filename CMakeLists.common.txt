
# KAI configuration
add_definitions(-DKAI_USE_CPP1Y)
add_definitions(-DKAI_USE_NAMESPACES -DKAI_NAMESPACE_NAME=kai)
add_definitions(-DKAI_DEBUG)
add_definitions(-DKAI_DEBUG_TRACE)
add_definitions(-DKAI_USE_EXCEPTIONS)

# don't need the Cellar/boost directory, except for OSX, 
# and doesn't hurt otherwise
include_directories($ENV{HOME}/Include $ENV{HOME}/dev/KAI/Include usr/local/Cellar/boost/1.60.0_2/include)

# add some default C++ compiler flags
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++1y -Wno-deprecation -fpermisssive")

# set output directories
set(dir $ENV{HOME})
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY "${dir}/lib")
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY "${dir}/lib")
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${dir}/bin")

set(HOME, $ENV{HOME})
set(KAI_ROOT, ${HOME}/dev/KAI)

link_directories({HOME}/lib)

set(KAI_VERSION_MAJOR, 0)
set(KAI_VERSION_MINOR, 1)
